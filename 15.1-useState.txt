//! useState() Hook

//! What is useState()?
--> useState() is a built-in Hook introduced in React 16.8.
--> It is used to create and manage State in Functional Components.
--> Whenever the State changes, React automatically re-renders the Component.

//*Definition:
--> useState() is a Hook that allows Functional Components to store and update dynamic data.

//! Syntax
const [state, setState] = useState(initialValue);

//! Explanation
state
--> Stores the current State value.

setState
--> Function used to update the State.

initialValue
--> Initial value assigned to the State.

//! Why useState()?
--> Store dynamic data.
--> Update the UI automatically.
--> Handle user interactions.
--> Replace State used in Class Components.

//! Import
import { useState } from "react";

//! Example 1: Number State
import { useState } from 'react'
const State = () => {
  let [count, setCount] = useState(0)
  let handleIncrement = () => {
    setCount(count + 1)
  }
  let handleDecrement = () => {
    setCount(count - 1)
  }
  let handleReset = () => {
    setCount(0)
  }
  return (
    <div>
      <h1>Count : {count}</h1>
      <button onClick={handleIncrement}>Increment</button>
      <button onClick={handleDecrement}>Decrement</button>
      <button onClick={handleReset}>Reset</button>
    </div>
  )
}
export default State

//! Flow
Initial State
      ↓
User Clicks Button
      ↓
setState()
      ↓
State Updated
      ↓
Component Re-rendered

//! Example 2: String State
import { useState } from 'react'
const State = () => {
  let [food, setFood] = useState("")
  let handleChickenBiryani = () => {
    setFood("Chicken Biryani 🐔")
  }
  let handleMuttonBiryani = () => {
    setFood("Mutton Biryani 🐐")
  }
  let handleVegBiryani = () => {
    setFood("Veg Biryani 🥔")
  }
  return (
    <div>
      <h1>I ate : {food}</h1>
      <button onClick={handleChickenBiryani}>Chicken Biryani</button>
      <button onClick={handleMuttonBiryani}>Mutton Biryani</button>
      <button onClick={handleVegBiryani}>Veg Biryani</button>
    </div>
  )
}
export default State

//! Example 3: Boolean State
import { useState } from 'react'
const State = () => {
  let [theme, setTheme] = useState(true)
  let handleTheme = () => {
    setTheme(!theme)
  }
  return (
    <div>
      <h1>Theme : {theme ? "Light" : "Dark"}</h1>
      <button onClick={handleTheme}>
        {theme ? "Switch to Dark Mode" : "Switch to Light Mode"}
      </button>
    </div>
  )
}
export default State

//! Example 4: Array State
import { useState } from 'react'
const State = () => {
  let [subjects, setSubjects] = useState(["Html", "Css", "Js"])
  let handleSubject = () => {
      setSubjects([...subjects,"React"])
  }
  return (
    <div>
      <h1>Frontend Subjects 😎</h1>
      <ol>
        {
          subjects.map((item,index) => {
            return <li key={index}>{item}</li>
          })
        }
      </ol>
      <button onClick={handleSubject}>Add React</button>
    </div>
  )
}
export default State

//*Note:
--> Spread Operator preserves existing properties.

//! Example 5: Object State
function App() {
    const [student, setStudent] = useState({
        name: "Rahul",
        age: 22
    });
    function updateAge() {
        setStudent({
            ...student,
            age: 23
        });

    }

    return (
        <>
            <h1>{student.name}</h1>
            <h2>{student.age}</h2>

            <button onClick={updateAge}>
                Update Age
            </button>
        </>
    );
}

//! Functional Updates
--> Use when the next State depends on the previous State.

❌ Normal Update
setCount(count + 1);

✔ Functional Update
setCount((prevCount) => prevCount + 1);

Advantage:
--> Always receives the latest State value.

//! Example
function Counter() {
    const [count, setCount] = useState(0);
    function increment() {
        setCount((prevCount) => prevCount + 1);
    }

    return (
        <>
            <h1>{count}</h1>

            <button onClick={increment}>
                Increment
            </button>
        </>
    );
}

//! useState() vs State in Class Components

| useState() | Class Component State |
|-------------|-----------------------|
| Functional Components | Class Components |
| useState() Hook | this.state |
| setState Function | this.setState() |
| No this Keyword | Uses this Keyword |
| Simpler Syntax | More Boilerplate |
