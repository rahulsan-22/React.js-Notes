//! Conditional Rendering in React.js

//! What is Conditional Rendering?
--> Conditional Rendering means displaying different UI based on a condition.
--> React allows us to render elements conditionally just like JavaScript conditions.
--> It is commonly used for:
    - Login/Logout UI
    - Loading States
    - User Roles
    - Showing/Hiding Components
    - Error Messages

Definition:
--> Conditional Rendering is the process of rendering different UI elements based on specific conditions.

//! Why Conditional Rendering?
--> Creates Dynamic UI.
--> Improves User Experience.
--> Displays relevant content based on user actions.
--> Avoids unnecessary UI rendering.

//! 1. Using Ternary Operator (Most Common)

//*Syntax:
condition ? expression1 : expression2

//*Example:
const ConditionalRendering = () => {
  let isAdmin = false;
  return <>
    {
      isAdmin ? <h1>Welcome Admin 😎</h1> : <h1>Welcome User 😁</h1>
    }
  </>
}
export default ConditionalRendering

//*Output:
Welcome User 😁

//! Example 2
import Hungry from "./Hungry"
import Sleep from "./Sleep"
const ConditionalRendering = () => {
  let feelingHungry = false
  return <>
    {
     feelingHungry ? <Hungry/> : <Sleep/>
    }
  </>
}
export default ConditionalRendering

//!Example 3 :
const ConditionalRendering = () => {
  let age = 15
  return (
    <div>
      {
        age>=18 ? <h1>Eligible to vote 🥳</h1> : <h1>Concentrate on your studies 🥺</h1>
      }
    </div>
  )
}
export default ConditionalRendering

//! Why Ternary Operator?
--> Most commonly used in React projects.
--> Simple and readable.
--> Directly works inside JSX.

//! 2. Using Logical AND (&&)

//*Syntax:
condition && expression

//*Note:
--> Expression renders only when condition is true.

//!Example 1 :
const ConditionalRendering = () => {
  let isAdmin = true
  return (
    <div>
      {isAdmin && <h1>Welcome Admin 🙏</h1>}
    </div>
  )
}
export default ConditionalRendering

//*Output:
Welcome Admin 🙏

//! Example 2:
import Sleep from './Sleep';
const ConditionalRendering = () => {
  let feelingSleepy = true;
  return (
    <div>
      {
        feelingSleepy && <Sleep/>
      }
    </div>
  )
}
export default ConditionalRendering

//*Output:
Sleep Component Rendered

//! When to Use && ?
--> When only True Condition UI is required.
--> No Else Condition needed.

//! 3. Using if-else Statement
--> if-else cannot be written directly inside JSX.
--> Use it before return.

//!Example 1 :
const ConditionalRendering = () => {
  let amount = 150;
  if (amount >= 200) {
    return <h1>Have Chicken Biryani 🐔</h1>
  } else {
    return <h1>Eat in PG 🥺</h1>
  }
}
export default ConditionalRendering

//!Example 2 : (Multiple Conditions)
const ConditionalRendering = () => {
  let marks = 10;
  if (marks >= 90) {
    return <h1>Grade A</h1>
  } else if (marks >= 60) {
    return <h1>Grade B</h1>
  } else {
    return <h1>Grade C</h1>
  }
}
export default ConditionalRendering

//! 4. Using Element Variables

//*Example:
const ConditionalRendering = () => {
  let headache = false
  let result;
  if (headache) {
    result = <h1>Have Coffee</h1>
  } else {
    result = <h1>Have Tea</h1>
  }
  return result
}
export default ConditionalRendering

//! Real-Time Example: Dark/Light Mode
const ConditionalRendering = () => {
  let darkMode = false
  return (
    <div>
      <button>{darkMode ? "Dark Mode" : "Light Mode"}</button>
    </div>
  )
}
export default ConditionalRendering

//! What Cannot Be Written Directly Inside JSX?
❌ Wrong:
return (
    <h1>
        {if(true){return "Hello"}}
    </h1>
);

//*Reason:
--> JSX accepts Expressions, not Statements.

✔ Correct:
const message = true ? "Hello" : "Bye";
return <h1>{message}</h1>;
