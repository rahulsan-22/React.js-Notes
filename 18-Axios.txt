//! Axios

//! What is Axios?
--> Axios is a popular JavaScript library used to make HTTP requests.
--> It is used to communicate with servers and APIs.
--> It allows us to fetch, send, update, and delete data easily.
--> Axios works with both Browser and Node.js environments.

//! Why Axios?
Before Axios
--> We use fetch() API.
--> Need to manually convert response to JSON.
--> More code.
--> Error handling is slightly more complex.

With Axios
--> Less Code.
--> Automatic JSON Conversion.
--> Better Error Handling.

//! API (Application Programming Interface)
--> API is a bridge between the Frontend and Backend.
--> It allows applications to exchange data.
--> Frontend sends requests.
--> Backend processes requests.
--> Backend sends responses.

//!Flow

React Application
      ↓
Axios Request
      ↓
Server / API
      ↓
Database
      ↓
Response
      ↓
React Application

//!HTTP Methods

//*Method	                        Purpose
GET	                              Fetch Data
POST	                           Create Data
PUT	                              Update Entire Data
PATCH	                           Update Partial Data
DELETE	                         Delete Data

//!POST REQUEST
import axios from "axios";

async function addProduct(){
 const product={
    title:"Laptop",
    price:50000
  };
  const response=await axios.post("https://dummyjson.com/products/add",product);
  console.log(response.data);
}

//!PUT REQUEST
async function updateProduct(){
const response=await axios.put("https://dummyjson.com/products/1",{title:"Gaming Laptop",price:70000});
console.log(response.data);
}

//!PATCH REQUEST
async function updateProduct(){
  const response=await axios.patch("https://dummyjson.com/products/1",{price:45000});
  console.log(response.data);
}

//!DELETE REQUEST
async function deleteProduct(){
  const response=await axios.delete("https://dummyjson.com/products/1");
  console.log(response.data);
}

//!COMPLETE GET REQUEST EXAMPLE:
import axios from 'axios'
import { useEffect, useState } from 'react'

const FetchData = () => {
  let [items, setItems] = useState(null)
  useEffect(() => {
    let api = async () => {
      let { data: { products } } = await axios.get("https://dummyjson.com/products")
      setItems(products)
    }
    api()
  }, [])
  return (
    <table border={1} rules="all" cellPadding={10} width="100%">
      <thead>
        <tr>
          <th>Sl.No.</th>
          <th>Title</th>
          <th>Category</th>
          <th>Description</th>
          <th>Image</th>
          <th>Price</th>
        </tr>
      </thead>
      <tbody>
        {
          items && items.map((product) => {
            return <tr key={product.id}>
              <td>{product.id}</td>
              <td>{product.title}</td>
              <td>{product.category}</td>
              <td>{product.description}</td>
              <td><img src={product.images[0]} alt={product.title} height="200" width="200"/></td>
              <td>${product.price}</td>
            </tr>
          })
        }
      </tbody>
    </table>
  )
}

export default FetchData
