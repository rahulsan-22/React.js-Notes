//! Adding Styles in React.js

//! What are Styles in React?
--> Styles are used to control the appearance of React Components.
--> React supports multiple ways to add CSS.
--> Different styling methods are suitable for different project requirements.

//!Definition:
--> Styling in React is the process of applying CSS to Components to improve their appearance and user experience.

//! Ways to Add Styles in React
1. Inline CSS
2. Global CSS
3. CSS Modules
4. Tailwind CSS

//!1) Inline CSS

//! What is Inline CSS?
--> Inline CSS means writing CSS directly inside JSX.
--> React uses JavaScript Objects to define styles.
--> Style names are written in camelCase.

//!Definition:
--> Inline CSS is a method of applying styles directly to JSX elements using the style attribute.

//! Syntax
style={{ property: "value" }}

//!Example:
function App() {
    return (
          <h1
            style={{
                color: "red",
                backgroundColor: "yellow",
                fontSize: "40px"
            }}
        >
            Hello Student
        </h1>
    );
}

//! Explanation
style
--> JSX attribute.

{{ }}
--> Outer {} → JavaScript Expression.
--> Inner {} → JavaScript Object.

//! Dynamic Styling
function App() {
    const isLoggedIn = true;
    return (
        <h1
            style={{
                color: isLoggedIn ? "green" : "red"
            }}
        >
            Welcome
        </h1>

    );
}

//!2) Global CSS

//! What is Global CSS?
--> Global CSS means writing styles in a separate CSS file.
--> These styles are available throughout the application.
--> Commonly used for application-wide styling.

//!Definition:
--> Global CSS is a stylesheet whose styles are shared across the entire React application.

//! Example

//*index.css
body{
    margin:0;
    font-family:Arial;
}

h1{
    color:red;
}

//! Import CSS
-->Import css either main.jsx or App.jsx
import "./index.css";

//! Component
function App() {
    return <h1>Hello Student</h1>;
}

//! 3) CSS Modules

//! What are CSS Modules?
--> CSS Modules provide locally scoped CSS.
--> Styles belong only to the Component that imports them.
--> Prevents class name conflicts.

//!Definition:
--> CSS Modules are CSS files where class names are scoped locally by default.

//! Creating CSS Module
FileName : Button.module.css

.button{
    background-color:blue;
    color:white;
}

//! Import

import styles from "./Button.module.css";

//! Usage
function Button(){
    return (
        <button className={styles.button}>
            Click
        </button>
    );
}

//! Generated Class Name
.button

↓

Button_button__A1b2X

--> Automatically generated by React.

//! Tailwind CSS

//! What is Tailwind CSS?
--> Tailwind CSS is a Utility-First CSS Framework.
--> Styles are applied using predefined utility classes.
--> No need to write traditional CSS for most styling.

//!Definition:
--> Tailwind CSS is a utility-first CSS framework used to rapidly build modern and responsive user interfaces.

//! Installation
npm install tailwindcss @tailwindcss/vite

//! Configure Vite

//*vite.config.js

import { defineConfig } from "vite";
import react from "@vitejs/plugin-react";
import tailwindcss from "@tailwindcss/vite";

export default defineConfig({
    plugins: [
        react(),
        tailwindcss(),
    ],
});

//! Import
index.css
@import "tailwindcss";

//! Example
function App(){
    return(
        <button
            className="
                bg-blue-500
                text-white
                px-5
                py-2
                rounded
            "
        >
            Click Me
        </button>
    );
}

//! Responsive Example
<div className="
    text-sm
    md:text-xl
    lg:text-3xl
">
Hello
</div>

//! Comparison
| Styling Method | Scope | Best Use |
|----------------|-------|----------|
| Inline CSS | Single Element | Dynamic Styles |
| Global CSS | Entire Application | Common Styles |
| CSS Modules | Component Level | Large Applications |
| Tailwind CSS | Utility Classes | Modern UI Development |
